129 void RendererData::RenderAndClearAllImageRequests()
130 {
131  std::stable_sort(image_draw_request_queue.begin(), image_draw_request_queue.end(), CompareImageRequests());
132  SDL_RenderSetScale(renderer, zoom_factor, zoom_factor);
133 
134  for (auto& request : image_draw_request_queue) {
135  glm::vec2 final_rendering_position = glm::vec2(request.x, request.y) - current_cam_pos;
136 
137  SDL_Texture* tex = GetImage(request.image_name);
138  SDL_Rect tex_rect;
139  SDL_QueryTexture(tex, NULL, NULL, &tex_rect.w, &tex_rect.h);
140 
141  // Apply scale
142  int flip_mode = SDL_FLIP_NONE;
143  if (request.scale_x < 0)
144  flip_mode |= SDL_FLIP_HORIZONTAL;
145  if (request.scale_y < 0)
146  flip_mode |= SDL_FLIP_VERTICAL;
147 
148  float x_scale = std::abs(request.scale_x);
149  float y_scale = std::abs(request.scale_y);
150 
151  tex_rect.w *= x_scale;
152  tex_rect.h *= y_scale;
153 
154  SDL_Point pivot_point = { static_cast<int>(request.pivot_x * tex_rect.w), static_cast<int>(request.pivot_y * tex_rect.h) };
155 
156  glm::ivec2 cam_dimensions = glm::ivec2(window_size.x, window_size.y);
157 
158  tex_rect.x = static_cast<int>(final_rendering_position.x * PIXELS_PER_METER + cam_dimensions.x * 0.5f * (1.0f / zoom_factor) - pivot_point.x);
159  tex_rect.y = static_cast<int>(final_rendering_position.y * PIXELS_PER_METER + cam_dimensions.y * 0.5f * (1.0f / zoom_factor) - pivot_point.y);
160 
161  // Apply tint / alpha to texture
162  SDL_SetTextureColorMod(tex, request.r, request.g, request.b);
163  SDL_SetTextureAlphaMod(tex, request.a);
164 
165  // Preform draw
166  SDL_RenderCopyEx(GetRenderer(), tex, NULL, &tex_rect, request.rotation_degrees,
167  &pivot_point, static_cast<SDL_RendererFlip>(flip_mode));
168 
169  // Remove tint / alpha from texture
170  SDL_SetTextureColorMod(tex, 255, 255, 255);
171  SDL_SetTextureAlphaMod(tex, 255);
172  }
173 
174  SDL_RenderSetScale(renderer, 1, 1);
175 
176  image_draw_request_queue.clear();
177 }
