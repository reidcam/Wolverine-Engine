25 void Input::ProcessEvent(const SDL_Event & e){
26 
27  if(e.type == SDL_KEYDOWN){
28  keyboard_states[e.key.keysym.scancode] = INPUT_STATE_JUST_BECAME_DOWN;
29  just_became_down_scancodes.push_back(e.key.keysym.scancode);
30  }
31  else if(e.type == SDL_KEYUP){
32  keyboard_states[e.key.keysym.scancode] = INPUT_STATE_JUST_BECAME_UP;
33  just_became_up_scancodes.push_back(e.key.keysym.scancode);
34  }
35 
36  // check for mouse events
37  if(e.type == SDL_MOUSEBUTTONDOWN){
38  int button;
39  if(e.button.button == SDL_BUTTON_LEFT){
40  button = 1;
41  }
42  else if(e.button.button == SDL_BUTTON_MIDDLE){
43  button = 2;
44  }
45  else if(e.button.button == SDL_BUTTON_RIGHT){
46  button = 3;
47  }
48  else{
49  button = 0;
50  }
51  mouse_states[button] = INPUT_STATE_JUST_BECAME_DOWN;
52  just_became_down_mouse_buttons.push_back(button);
53  }
54  else if(e.type == SDL_MOUSEBUTTONUP){
55  int button;
56  if(e.button.button == SDL_BUTTON_LEFT){
57  button = 1;
58  }
59  else if(e.button.button == SDL_BUTTON_MIDDLE){
60  button = 2;
61  }
62  else if(e.button.button == SDL_BUTTON_RIGHT){
63  button = 3;
64  }
65  else{
66  button = 0;
67  }
68  mouse_states[button] = INPUT_STATE_JUST_BECAME_UP;
69  just_became_up_mouse_buttons.push_back(button);
70  }
71 
72  // check for mouse scroll
73  if(e.type == SDL_MOUSEWHEEL){
74  mouse_scroll_this_frame = e.wheel.preciseY;
75  }
76 
77  // check for mouse movement

80  }

104  }
