15 void CollisionDetector::BeginContact(b2Contact* contact)
16 {
17  b2Fixture* fixture_a = contact->GetFixtureA();
18  b2Fixture* fixture_b = contact->GetFixtureB();
19  uintptr_t userDataA = fixture_a->GetUserData().pointer;
20  Actor* actor_a = reinterpret_cast<Actor*>(userDataA);
21  uintptr_t userDataB = fixture_b->GetUserData().pointer;
22  Actor* actor_b = reinterpret_cast<Actor*>(userDataB);
23 
24  // one of the colliders is a phantom fixture
25  if (actor_a == nullptr || actor_b == nullptr)
26  return;
27 
28  b2WorldManifold world_manifold;
29  contact->GetWorldManifold(&world_manifold);
30 
31  Collision collision;
32  collision.point = world_manifold.points[0]; // get the first point of collision
33 
34  b2Vec2 relative_velocity = fixture_a->GetBody()->GetLinearVelocity() - fixture_b->GetBody()->GetLinearVelocity();
35  collision.relative_velocity = relative_velocity;
36 
37  collision.normal = world_manifold.normal;
38 
39  collision.other_actor_id = actor_b->ID;
40  collision.this_actor_id = actor_a->ID;
41 
42  // determine if we have 2 triggers or 2 colliders
43  if (fixture_a->IsSensor() && fixture_b->IsSensor()) {
44  collision.point = b2Vec2(-999.0f, -999.0f); // not valid for trigger
45  collision.normal = b2Vec2(-999.0f, -999.0f); // not valid for trigger
46  OnEnter::OnTriggerEnter(collision);
47 
48  collision.other_actor_id = actor_a->ID;
49  collision.this_actor_id = actor_b->ID;
50 
51  OnEnter::OnTriggerEnter(collision);
52  }
53  else if (!fixture_a->IsSensor() && !fixture_b->IsSensor()) {
54  OnEnter::OnCollisionEnter(collision);
55 
56  collision.other_actor_id = actor_a->ID;
57  collision.this_actor_id = actor_b->ID;
58 
59  OnEnter::OnCollisionEnter(collision);
60  }

116 void OnEnter::OnTriggerEnter(const Collision& collision)
117 {
118  size_t number_of_components = Actors::GetNumberOfComponents(collision.this_actor_id);
119 
120  for (size_t i = 0; i < number_of_components; i++) {
121  try {
122  sol::table component = Actors::GetComponentByIndex(collision.this_actor_id, i);
123  sol::function OnTriggerEnter = component["OnTriggerEnter"];
124 
125  // If this component is dead, skip it
126  if ((component)["REMOVED_FROM_ACTOR"] == true)
127  {
128  continue;
129  }
130 
131  // Skip this component if the actor or component aren't enabled
132  if (!Actors::GetActorEnabled(collision.this_actor_id) || (component)["enabled"] == false)
133  {
134  continue;
135  }
136 
137  if (OnTriggerEnter.valid())
138  {
139  OnTriggerEnter(component, collision);
140  }
141  }
142  catch (const sol::error& e) {
143  std::cerr << "Caught Lua runtime error: " << e.what() << std::endl;
144  }
145  catch (const std::exception& e) {
146  std::cerr << "Caught C++ exception: " << e.what() << std::endl;
147  }
148  catch (...) {
149  std::cerr << "Caught unknown exception." << std::endl;
150  }
151  }
