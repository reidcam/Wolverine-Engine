30 void Actors::Cleanup()
31 {
32  // Erase the dead actors from every vector
33  int i = 0;
34  while (i < num_loaded_actors)
35  {
36  if (IDs[i] == -1)
37  {
38  // Erase from the vectors
39  IDs.erase(IDs.begin() + i);
40  names.erase(names.begin() + i);
41  actor_enabled.erase(actor_enabled.begin() + i);
42  components.erase(components.begin() + i);
43  }
44  else
45  {
46  i++;
47  }
48  }
49 
50  // Update the id_to_index map with new indexes for living ids
51  for (int i = 0; i < IDs.size(); i++)
52  {
53  id_to_index[IDs[i]] = i;
54  }
55 }

60 void Actors::ProcessAddedComponents()
61 {
62  // Stores the components that didn't get processed this frame, so we can re-add
63  // them to "components_to_init" when we've finished processing the others
64  std::vector<std::shared_ptr<sol::table>> not_processed;
65 
66  for (auto& component : components_to_init)
67  {
68  int actor_index = GetIndex((*component)["actor"]["ID"]);
69 
70  // If this component has been removed, skip it
71  if ((*component)["REMOVED_FROM_ACTOR"] == true)
72  {
73  continue;
74  }
75 
76  // Skip this component if the actor or component aren't enabled
77  if (!actor_enabled[actor_index] || (*component)["enabled"] == false)
78  {
79  not_processed.push_back(component);
80  continue;
81  }
82 
83  // Add to the appropriate lifecycle functions list so that they will start being called by the engine.
84  sol::function OnUpdate = (*component)["OnUpdate"];
85  if (OnUpdate.valid()) { components_to_update.push_back(component); }
86  sol::function OnLateUpdate = (*component)["OnLateUpdate"];
87  if (OnLateUpdate.valid()) { components_to_update_late.push_back(component); }
88 
89  // Call "OnStart" if it exists for this component
90  try
91  {
92  // OnStart is called for each component the frame they are loaded into the game
93  sol::function OnStart = (*component)["OnStart"];
94  if (OnStart.valid())
95  {
96  OnStart(*component);
97  }
98  }
99  catch(const std::exception& e)
100  {
101  std::string errorMessage = e.what();
102 #ifdef _WIN32
103  std::replace(errorMessage.begin(), errorMessage.end(), '\\', '/');
104 #endif
105  std::cout << "\033[31m" << names[actor_index] << " : " << errorMessage << "\033[0m" << std::endl;
106  }
107  }
108 
109  // Remove all the processed components from the list
110  components_to_init.clear();
111  components_to_init = not_processed;
112 }

297 int Actors::LoadActorWithJSON(const rapidjson::Value& actor_data)
298 {
299  // Gives actor their ID
300  IDs.push_back(num_total_actors);
301  int index = (int)IDs.size() - 1;
302  id_to_index[num_total_actors] = index;
303 
304  // Assigns the values to the new actor
305  if (actor_data.HasMember("name"))
306  {
307  names.push_back(actor_data["name"].GetString());
308  }
309  else
310  {
311  // Gives this actor a default name if none is specified for it
312  names.push_back("ActorName");
313  }
314 
315  if (actor_data.HasMember("enabled"))
316  {
317  actor_enabled.push_back(actor_data["enabled"].GetBool());
318  }
319  else
320  {
321  // Sets the actor to be enabled by default
322  actor_enabled.push_back(true);
323  }
324 
325  //-------------------------------------------------------
326  // Components
327 
328  // Stores all of the created components
329  std::vector<std::shared_ptr<sol::table>> new_components_list;
330  if (actor_data.HasMember("components"))
331  {
332  const rapidjson::Value& actor_components = actor_data["components"];
333 
334  // Iterate over each component
335  for (rapidjson::Value::ConstMemberIterator itr = actor_components.MemberBegin(); itr != actor_components.MemberEnd(); itr++)
336  {
337  // Creates and gets a reference to a new table on the Lua stack
338  sol::table new_component = LuaAPI::GetLuaState()->create_table();
339 
340  // The key of this component
341  std::string key = itr->name.GetString();
342 
343  // Establishes inheritance between the new component and its type if specified
344  if (itr->value.HasMember("type"))
345  {
346  std::string type = itr->value["type"].GetString();
347 
348  // Establishes our new component according to its type
349  if (ComponentManager::IsComponentTypeNative(type))
350  {
351  new_component = ComponentManager::NewNativeComponent(type);
352  }
353  else
354  {
355  ComponentManager::EstablishInheritance(new_component, *GetComponentType(type));
356  }
357 
358  // Allows the component to know its own type
359  new_component["type"] = type;
360  // Gives the component its key
361  new_component["key"] = key;
362  // Sets the component to be enabled by default
363  new_component["enabled"] = true;
364  }
365  // If the type for this component is not specified anywhere, throw an error
366  else
367  {
368  std::cout << "error: component type unspecified for " << key << " on " << names[names.size() - 1];
369  exit(0);
370  }
371 
372  //-------------------------------------------------------
373  // Preform required overrides on component properties
374  // Sets component properties to specified values
375  const rapidjson::Value& component_properties = itr->value;
376  for (rapidjson::Value::ConstMemberIterator itr2 = component_properties.MemberBegin(); itr2 != component_properties.MemberEnd(); itr2++)
377  {
378  std::string property_name = itr2->name.GetString();
379  sol::type property_type = new_component[property_name].get_type();
380 
381  sol::lua_value property = "";
382 
383  JsonToLuaObject(property, itr2->value, property_type);
384 
385  new_component[property_name] = property;
386  }
387 
388  //-------------------------------------------------------
389  // Injects the new component with a reference to its actor
390  Actor* _a = new Actor();
391  _a->ID = num_total_actors;
392  new_component["actor"] = _a;
393 
394  // Add the new component to the "components_to_init" and "components" vectors
395  std::shared_ptr<sol::table> ptr = std::make_shared<sol::table>(new_component);
396  components_to_init.push_back(ptr);
397  new_components_list.push_back(ptr);
398  }
399  }
400  components.push_back(new_components_list);
401 
402  // Update the number of loaded actors.
403  num_loaded_actors++;
404  num_total_actors++;
405 
406  return num_total_actors - 1;
407 }

417 void Actors::JsonToLuaObject(sol::lua_value& value, const rapidjson::Value& data, sol::type type)
418 {
419  // Adds the property to the component
420  if (type == sol::type::string) { value = data.GetString(); }
421  else if (type == sol::type::boolean) { value = data.GetBool(); }
422  else if (type == sol::type::number)
423  {
424  if (data.IsDouble()) { value = data.GetDouble(); }
425  else { value = data.GetInt(); }
426  }
427  else if (type == sol::type::table)
428  {
429  sol::table _table = LuaAPI::GetLuaState()->create_table();
430  _table[0] = sol::object(*LuaAPI::GetLuaState());
431 
432  // Add all of the values in the data to our new table.
433  int i = 1;
434  for (rapidjson::Value::ConstMemberIterator itr = data.MemberBegin(); itr != data.MemberEnd(); itr++)
435  {
436  const rapidjson::Value& _data = itr->value;
437 
438  sol::lua_value _value = "";
439 
440  // Determine the type of value that is being added to the table
441  sol::type _type = sol::type::none;
442  if (itr->value.IsString()) { _type = sol::type::string; }
443  else if (itr->value.IsBool()) { _type = sol::type::boolean; }
444  else if (itr->value.IsInt() || itr->value.IsDouble()) { _type = sol::type::number; }
445  else if (itr->value.IsObject()) { _type = sol::type::table; }
446 
447  JsonToLuaObject(_value, _data, _type);
448 
449  // Add the value to the new table
450  _table[i] = _value;
451  i++;
452  }
453  value = _table;
454  }
455  else
456  {
457  // If the sol type is not any of the above, determine the object type based off of the json
458  sol::type second_type;
459  if (data.IsString()) { second_type = sol::type::string; }
460  else if (data.IsBool()) { second_type = sol::type::boolean; }
461  else if (data.IsNumber()) { second_type = sol::type::number; }
462  else if (data.IsObject()) { second_type = sol::type::table; }
463  JsonToLuaObject(value, data, second_type);
464  }
465 }

531 void Actors::RemoveComponentFromActor(int actor_id, sol::table component)
532 {
533  if (!component.valid())
534  {
535  return;
536  }
537 
538  int actor_index = GetIndex(actor_id);
539  if (actor_index == -1) {return;}
540 
541  // Removes this component from the actor
542  int component_index = -1;
543  for (int i = 0; i < components[actor_index].size(); i++) // Find the index of this component
544  {
545  if ((*components[actor_index][i]).pointer() == component.pointer())
546  {
547  (component)["REMOVED_FROM_ACTOR"] = true;
548  components_to_delete.push(std::make_shared<sol::table>(component));
549  component_index = i;
550  break;
551  }
552  }
553  if (component_index != -1) {
554  components[actor_index].erase(components[actor_index].begin() + component_index);
555  } // remove the component from the actor
556 }
