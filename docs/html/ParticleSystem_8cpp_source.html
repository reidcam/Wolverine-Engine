64 void ParticleSystem::UpdateParticle(Particle* particle)
65 {
66  // Update particles according to a pattern (also update particle's hitbox size to match if size changes)
67 
68  // Update color
69  if (change_color)
70  {
71  // The change in color per frame
72  std::vector<float> delta_color = GetDeltaColor(particle);
73 
74  particle->color[0] += delta_color[0];
75  if (particle->color[0] >= 255.0f) {particle->color[0] = 255.0f;}
76  if (particle->color[0] <= 0.0f) {particle->color[0] = 0.0f;}
77 
78  particle->color[1] += delta_color[1];
79  if (particle->color[1] >= 255.0f) {particle->color[1] = 255.0f;}
80  if (particle->color[1] <= 0.0f) {particle->color[1] = 0.0f;}
81 
82  particle->color[2] += delta_color[2];
83  if (particle->color[2] >= 255.0f) {particle->color[2] = 255.0f;}
84  if (particle->color[2] <= 0.0f) {particle->color[2] = 0.0f;}
85 
86  particle->color[3] += delta_color[3];
87  if (particle->color[3] >= 255.0f) {particle->color[3] = 255.0f;}
88  if (particle->color[3] <= 0.0f) {particle->color[3] = 0.0f;}
89  }
90 
91  // Update size
92  if (change_size)
93  {
94  float delta_size = 0.0f;
95 
96  // Alters the size based on the pattern chosen
97  switch (size_pattern[0])
98  {
99  case 'l':
100  // Linear pattern
101  delta_size = size_change_per_second / 60;
102  particle->size += delta_size;
103  break;
104 
105  case 'e':
106  // Exponential pattern
107  delta_size = std::pow(particle->age / particle_lifetime, 3.0f);
108  particle->size += delta_size * size_change_per_second;
109  break;
110 
111  case 's':
112  // Sine pattern
113  delta_size = std::sin((particle->age / size_sine_loop) * (2.0f * b2_pi));
114  particle->size = std::abs(delta_size * starting_size);
115  break;
116 
117  default:
118  break;
119  }
120 
121  // Prevents particle from overflowing
122  if (particle->size < 0.0f)
123  {
124  particle->size = -0.001f;
125  }
126 
127  // Update hitbox to match the new size
128  if (has_collider)
129  {
130  particle->body->DestroyFixture(particle->body->GetFixtureList());
131  particle->body->CreateFixture(GetNewCollider(particle->size));
132  }
133  }
134 
135  // Update movement
136  if (change_movement)
137  {
138  b2Vec2 velocity = b2Vec2(cos(particle->direction * (b2_pi / 180.0f)) * speed, sin(particle->direction * (b2_pi / 180.0f)) * speed);
139  b2Vec2 newVelocity = velocity;
140  float delta_speed;
141 
142  // Alters the movement based on the pattern chosen
143  switch (movement_pattern[0])
144  {
145  case 's':
146  // Sine pattern
147  delta_speed = std::sin((particle->age / speed_sine_loop) * (2.0f * b2_pi)) * speed_sine_amplitude;
148 
149  velocity = b2Vec2(-sin(particle->direction * (b2_pi / 180.0f)) * speed, cos(particle->direction * (b2_pi / 180.0f)) * speed);
150  velocity.Normalize();
151  velocity *= delta_speed;
152 
153  newVelocity += velocity;
154  break;
155 
156  case 'e':
157  break;
158 
159  default:
160  break;
161  }
162 
163  particle->body->SetLinearVelocity(newVelocity);
164  }
165 }

167 std::vector<float> ParticleSystem::GetDeltaColor(Particle* particle)
168 {
169  float percent_of_lifetime = particle->age / particle_lifetime;
170 
171  // Find the future and former colors
172  if (colors.size() > 1)
173  {
174  if (percent_of_lifetime > colors[particle->future_color_index][4] / 100)
175  {
176  if (particle->future_color_index + 1 < colors.size())
177  {
178  particle->future_color_index++;
179  }
180  particle->former_color_index++;
181  if (particle->former_color_index == particle->future_color_index)
182  {
183  particle->former_color_index--;
184  }
185  }
186  }
187  else
188  {
189  return {0, 0, 0, 0};
190  }
191 
192 
193  // The time that it takes to transition from the old color to the new one in seconds:
194  float transition_time = (std::fabs(colors[particle->future_color_index][4] - colors[particle->former_color_index][4]) / 100) * particle_lifetime;
195 
196  // Calculates the change in the RGB and A values.
197  std::vector<float> delta_color;
198  for (int i = 0; i < 4; i++)
199  {
200  delta_color.push_back((colors[particle->future_color_index][i] - colors[particle->former_color_index][i]) / (60 * transition_time));
201  }
202  return delta_color;
203 }

346 void ParticleSystem::OnUpdate()
347 {
348  /* Emission Phase */
349  if (emitting && num_particles < MAX_NUM_PARTICLES)
350  {
351  // Emission rate is impercise with larger rates,
352  // But the larger rates should result in a more accurate aproximation
353  if (emission_rate > 60)
354  {
355  int num_particles_to_spawn = std::ceil(EngineUtils::RandomNumber(0, (emission_rate / 60.0f) * 2u, 100));
356  //std::cout << num_particles_to_spawn << std::endl;
357  if (num_particles + num_particles_to_spawn < MAX_NUM_PARTICLES)
358  {
359  for (int i = 0; i < num_particles_to_spawn; i++)
360  {
361  CreateParticle();
362  }
363  }
364  }
365  // Emission rate is exact when less than 60, this is because each second is exactly 60 frames long.
366  else if (framesActive % (60 / emission_rate) == 0)
367  {
368  CreateParticle();
369  }
370  }
371  if (num_particles == 0) {return;}
372 
373  std::queue<Particle*> renderingQueue;
374 
375  /* Simulation Phase */
376  // Physics done automatically by box2d
377  while (!particles.empty())
378  {
379  Particle* particle = particles.front();
380  particles.pop();
381 
382  UpdateParticle(particle);
383 
384  // Ages the particle by 1 frame
385  particle->age += (1.0f / 60.0f);
386 
387  // Removes a particle from the queue if it's lifetime is up
388  if (particle->age < particle_lifetime && particle->size >= 0.0f)
389  {
390  renderingQueue.push(particle);
391  }
392  else
393  {
394  DestroyParticle(particle);
395  }
396  }
397 
398  /* Rendering Phase */
399  while (!renderingQueue.empty())
400  {
401  Particle* particle = renderingQueue.front();
402  renderingQueue.pop();
403  particles.push(particle);
404 
405  RenderParticle(particle);
406  }
407 
408  // If this particle system has been emitting for longer than its duration then stop emitting.
409  if (!loop && framesActive / 60.0f > duration)
410  {
411  emitting = false;
412  }
413  else
414  {
415  framesActive++;
416  }
417 }
