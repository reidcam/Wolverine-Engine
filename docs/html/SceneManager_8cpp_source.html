30 void Scene::UpdateActors()
31 {
32  current_scene_lifetime++;
33 
34  // Cleans up all of the dead actors in the actor database
35  if (current_scene_lifetime % 60 == 0)
36  {
37  Actors::Cleanup();
38  }
39 
40  // Process all of the components added to actors on the previous frame
41  Actors::ProcessAddedComponents();
42 
43  // Update all actors
44  Actors::Update();
45 
46  // Late update all actors
47  Actors::LateUpdate();
48 
49  // Processes all of the components removed from actors this frame
50  Actors::ProcessRemovedComponents();
51 
52  // Destroys all of the dead actors
53  // TODO: Take another look at how to remove dead actors from a scene, this feels VERY slow
54  for (auto actor : dead_actors)
55  {
56  // Destroy all of the actors that have been prepped for destruction.
57  Actors::DestroyActor(actor);
58 
59  int index_to_remove = -1;
60  // Find the index of the actor within 'actors' and delete it
61  for (int i = 0; i < actors.size(); i++)
62  {
63  if (actor == actors[i])
64  {
65  index_to_remove = i;
66  break;
67  }
68  }
69  // Erase the dead actor
70  if (index_to_remove != -1) { actors.erase(actors.begin() + index_to_remove); }
71  }
72  dead_actors.clear();
73 }

105 void Scene::LoadNewScene()
106 {
107  load_new_scene = false;
108  current_scene_name = new_scene_name;
109  current_scene_lifetime = 0;
110 
111  rapidjson::Document scene_document;
112  EngineUtils::ReadJsonFile(GetScenePath(new_scene_name), scene_document);
113 
114  if (scene_document.HasMember("actors"))
115  {
116  // Load all of the new actors into the game
117  for (auto& member : (scene_document)["actors"].GetArray())
118  {
119  // If the actor has a value for template, combine the actor with the template before loading it
120  if (member.HasMember("template"))
121  {
122  rapidjson::Document combined_actor;
123 
124  rapidjson::Document lhs;
125  lhs.CopyFrom(member, lhs.GetAllocator());
126  rapidjson::Document rhs;
127  rhs.CopyFrom(*GetTemplate(member["template"].GetString()), rhs.GetAllocator());
128 
129  EngineUtils::CombineJsonDocuments(lhs, rhs, combined_actor);
130 
131  // FOR TESTS: Output the combined JSON as a string
132  // rapidjson::StringBuffer buffer;
133  // rapidjson::PrettyWriter<rapidjson::StringBuffer> writer(buffer);
134  // combined_actor.Accept(writer);
135  // std::cout << buffer.GetString() << std::endl;
136 
137  actors.push_back(Actors::LoadActorWithJSON(combined_actor));
138  }
139  else
140  {
141  actors.push_back(Actors::LoadActorWithJSON(member));
142  }
143  }
144  }
145  // Return after loading the needed scene
146  return;
147 }
