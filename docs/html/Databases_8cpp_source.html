234 
235  // Fills up the database if the path exists

244  sol::load_result script = LuaAPI::GetLuaState()->load_file(file.path().string().c_str());
245  if (script.valid())

247  // Load the script into the lua state

249

252 
253  // Load the component type into our database
254  loaded_component_types.insert(
255  {
256  type_name,

260  }
261  else
